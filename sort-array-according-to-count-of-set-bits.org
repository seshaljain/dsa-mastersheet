#+TITLE: Sort array according to count of set bits
#+AUTHOR: Seshal Jain
#+TAGS[]: search_sort

#+begin_src cpp
class Solution {
public:
  int findSetBits(int n) {
    int bCount = 0;

    while (n != 0) {
      n &= (n - 1);
      bCount++;
    }

    return bCount;
  }

  void sortBySetBitCount(int arr[], int n) {
    stable_sort(arr, arr + n, [&](int a, int b) -> bool {
      return findSetBits(a) > findSetBits(b);
    });
  }
};
#+end_src
